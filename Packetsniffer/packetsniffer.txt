
import jpcap.JpcapCaptor;
import jpcap.NetworkInterface;
import jpcap.packet.Packet;

public class PacketSniffer {
    public static void main(String[] args) {
        try {
       
            NetworkInterface[] devices = JpcapCaptor.getDeviceList();

            System.out.println("Available Network Devices:");
            for (int i = 0; i < devices.length; i++) {
                System.out.println(i + ": " + devices[i].name + " - " + devices[i].description);
            }

       
            NetworkInterface device = devices[0];

            JpcapCaptor captor = JpcapCaptor.openDevice(device, 65535, true, 20);

            System.out.println("\nCapturing packets...\n");

           
            while (true) {
                Packet packet = captor.getPacket();

                if (packet != null) {
                    System.out.println("Packet Captured:");
                    System.out.println("Length: " + packet.len);
                    System.out.println("Data: " + packet.toString());
                    System.out.println("------------------------");
                }
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
